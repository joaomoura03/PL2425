program TestaContarVogais;

function ContarVogais(Texto: string): integer;
var
    I: integer;
    C: char;
begin
    ContarVogais := 0;
    for I := 1 to Length(Texto) do
    begin
        C := UpCase(Texto[I]);
        if C in ['A', 'E', 'I', 'O', 'U'] then
            Inc(ContarVogais);
    end;
end;

var
  Frase: string;

begin
  writeln('Digite um texto:');

end.
